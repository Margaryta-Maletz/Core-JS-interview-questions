Деструктуризация

из объекта data извлечь locale и url


							const data = {
								title: "Page title",
								url: "localhost",
								translations: {
									country: "US",
									locale:"en-us"
								},
							}
						

							let { url, translation: { locale } } = data;
						

							let { url: newUrl, translation: { locale } } = data;
						

							let { url: newUrl = 'local', translation: { locale } } = data;

Написать функцию для нахождения суммы двух чисел

							function sum (num1, num2) {
							    return num1 + num2;
							}
							sum(1, 2); // 3
Если передаем только один параметр, то вернуть его значение

							function sum (num1, num2) {
							    return num1 + num2;
							}
							sum(1, 2); // 3
							sum(1); // NaN
Если передаем только один параметр, то вернуть его значение

							function sum (num1, num2 = 0) {
							    return num1 + num2;
							}
							sum(1, 2); // 3
							sum(1); // 1
