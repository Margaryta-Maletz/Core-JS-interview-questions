Деструктуризация объекта

из объекта data извлечь url и locale


							const data = {
								title: "Page title",
								url: "localhost",
								translations: {
									country: "US",
									locale:"en-us"
								},
							}
						

							let { url, translation: { locale } } = data;

							console.log(url); // "localhost"
							console.log(locale); // "en-us"

Деструктуризация объекта

из объекта data извлечь url и locale


							const data = {

Деструктуризация объекта

из объекта data извлечь title и все остальные параметры сохранить в rest


							const data = {
								title: "Page title",
								url: "localhost",
								translations: {
									country: "US",
									locale:"en-us"
								},
							}
						

							let { title, ...rest } = data;

							console.log(title); // "Page title"
							console.log(rest); // "objectObject"
							console.log(rest.url); // "localhost"
							console.log(rest.translations.locale); // "en-us"

						

							rest.translations.locale = "ru";
							console.log(data.translations.locale); // "ru"
