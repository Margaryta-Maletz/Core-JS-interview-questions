Currying


							function curry(f) { // curry(f) выполняет каррирование
  								return function(a) {
    								return function(b) {
      									return f(a, b);
    								};
  								};
							}

							// использование
							function sum(a, b) {
  								return a + b;
							}

							let curriedSum = curry(sum);

							alert( curriedSum(1)(2) ); // 3

Каррирование https://learn.javascript.ru/.

Recursion


							function pow(x, n) {
  								if (n == 1) {
    								return x;
  								} else {
    								return x * pow(x, n - 1);
  								}
							}

							alert( pow(2, 3) ); // 8
